105 template <typename string_type_t>
106 struct basic_kerning_style : public basic_full_width_style<string_type_t>
107 {
108 public: // public type definition
109 using string_type = string_type_t; // String Type
110 using char_type = typename string_type_t::value_type; // Character Type
111 using traits_type = typename string_type_t::traits_type; // Traits Type
112 using size_type = typename string_type_t::size_type; // Size Type
113
114 using fig_char_type = std::vector<string_type_t>; // Figlet char
115 using fig_str_type = std::vector<string_type_t>; // Figlet String
116
117 private: // private typedefs
118 using sstream_type = std::basic_stringstream<char_type>; // Sstream Type
119 using ostream_type = std::basic_ostream<char_type>; // Ostream Type
120 using istream_type = std::basic_istream<char_type>; // Istream Type
121 using ifstream_type = std::basic_ifstream<char_type>; // Ifstream Type
122 using ofstream_type = std::basic_ofstream<char_type>; // Ofstream Type
123
124 protected: // protected methods
128 void trim_fig_str_and_fig_char(fig_str_type &fig_str, fig_char_type &fig_chr) const
129 {
130 // left spaces and right spaces
131 std::vector<size_type> elem;
132
133 // count space
134 for (size_type i = 0; i < fig_str.size(); ++i)
135 {
136 int l_count = 0, r_count = 0;
137
138 for (auto itr = fig_str[i].rbegin(); itr != fig_str[i].rend(); ++itr)
139 {
140 if (*itr == ' ')
141 ++l_count;
142 else
143 break;
144 }
145
146 for (auto itr = fig_chr[i].begin(); itr != fig_chr[i].end(); ++itr)
147 {
148 if (*itr == ' ')
149 ++r_count;
150 else
151 break;
152 }
153
154 elem.push_back(l_count + r_count);
155 }
156
157 // minimum
158 const auto min = *std::min_element(elem.begin(), elem.end());
159
160 // for each line
161 for (size_type i = 0; i < fig_str.size(); ++i)
162 {
163 size_type siz = min + 1;
164
165 while (--siz > 0 && fig_str[i].back() == ' ')
166 {
167 fig_str[i].pop_back();
168 }
169
170 fig_chr[i].erase(0, siz);
171 }
172 }
173
174 public: // Public overrides
178 fig_str_type get_fig_str(std::vector<fig_char_type> fig_chrs) const override
179 {
180 // fig str container type
181 fig_str_type fig_str(this->height);
182
183 // verify height
184 this->verify_height(fig_chrs);
185
186 // for each fig char
187 for (auto &fig_chr : fig_chrs)
188 {
189 this->trim_fig_str_and_fig_char(fig_str, fig_chr);
190 this->add_fig_str_and_fig_char(fig_str, fig_chr);
191 }
192
193 // return
194 return this->rm_hardblank(fig_str);
195 }
196
200 shrink_type get_shrink_level() const override
201 {
202 return shrink_type::KERNING;
203 }

205 public: // static methods
209 static auto make_shared()
210 {
211 return std::make_shared<basic_kerning_style>();
212 }

241 auto smush_rules(char_type lc, char_type rc) const
242 {
243 //()
244 if (lc == traits_type::to_char_type(' '))
245 {
246 return rc;
247 }
248
249 if (rc == traits_type::to_char_type(' '))
250 {
251 return lc;
252 }
253
254 //(Equal character smush)
255 if (lc == rc)
256 {
257 return rc;
258 }
259
260 //(Underscores smush)
261 if (lc == traits_type::to_char_type('_') && cvt<string_type>("|/\\[]{}()<>").find(rc) != string_type::npos)
262 {
263 return rc;
264 }
265
266 if (rc == traits_type::to_char_type('_') && cvt<string_type>("|/\\[]{}()<>").find(lc) != string_type::npos)
267 {
268 return lc;
269 }
270
271 //(Hierarchy Smushing)
272 auto find_class = [](char_type ch)
273 {
274 if (ch == traits_type::to_char_type('|'))
275 {
276 return 1;
277 }
278
279 if (ch == traits_type::to_char_type('/') || ch == traits_type::to_char_type('\\'))
280 {
281 return 3;
282 }
283
284 if (ch == traits_type::to_char_type('[') || ch == traits_type::to_char_type(']'))
285 {
286 return 4;
287 }
288
289 if (ch == traits_type::to_char_type('{') || ch == traits_type::to_char_type('}'))
290 {
291 return 5;
292 }
293
294 if (ch == traits_type::to_char_type('(') || ch == traits_type::to_char_type(')'))
295 {
296 return 6;
297 }
298
299 return 0;
300 };
301
302 size_type c_lc = find_class(lc);
303 size_type c_rc = find_class(rc);
304
305 if (c_lc > c_rc)
306 {
307 return lc;
308 }
309
310 if (c_rc > c_lc)
311 {
312 return rc;
313 }
314
315 //(Opposite smush)
316 if (lc == traits_type::to_char_type('[') && rc == traits_type::to_char_type(']'))
317 {
318 return traits_type::to_char_type('|');
319 }
320
321 if (lc == traits_type::to_char_type(']') && rc == traits_type::to_char_type('['))
322 {
323 return traits_type::to_char_type('|');
324 }
325
326 if (lc == traits_type::to_char_type('{') && rc == traits_type::to_char_type('}'))
327 {
328 return traits_type::to_char_type('|');
329 }
330
331 if (lc == traits_type::to_char_type('}') && rc == traits_type::to_char_type('{'))
332 {
333 return traits_type::to_char_type('|');
334 }
335
336 if (lc == traits_type::to_char_type('(') && rc == traits_type::to_char_type(')'))
337 {
338 return traits_type::to_char_type('|');
339 }
340
341 if (lc == traits_type::to_char_type(')') && rc == traits_type::to_char_type('('))
342 {
343 return traits_type::to_char_type('|');
344 }
345
346 //(Big X smush)
347 if (lc == traits_type::to_char_type('/') && rc == traits_type::to_char_type('\\'))
348 {
349 return traits_type::to_char_type('|');
350 }
351
352 if (lc == traits_type::to_char_type('\\') && rc == traits_type::to_char_type('/'))
353 {
354 return traits_type::to_char_type('Y');
355 }

358 {
359 return traits_type::to_char_type('X');
360 }
361
362 //(universal smush)
363 return lc;

369 void smush_fig_str_and_fig_char(fig_str_type &fig_str, fig_char_type fig_chr) const
370 {
371 // determine if smushable if not the just add and return
372 for (size_type i = 0; i < this->height; ++i)
373 {
374 if ((fig_str[i].back() == this->hard_blank) && !(fig_chr[i].front() == this->hard_blank))
375 {
376 return this->add_fig_str_and_fig_char(fig_str, fig_chr);
377 }
378 else if (fig_str[i].size() == 0 || fig_str[i].size() == 0)
379 {
380 return this->add_fig_str_and_fig_char(fig_str, fig_chr);
381 }
382 }
383
384 // smush the fig str and fig char
385 for (size_type i = 0; i < fig_str.size(); ++i)
386 {
387 fig_str[i].back() = this->smush_rules(fig_str[i].back(), fig_chr[i].front());
388 fig_chr[i].erase(fig_chr[i].begin());
389 }
390
391 // Add the fig char to the fig str
392 this->add_fig_str_and_fig_char(fig_str, fig_chr);
393 }
