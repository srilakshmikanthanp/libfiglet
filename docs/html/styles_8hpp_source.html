101 {
102 public: // public type definition
103 using string_type = string_type_t; // String Type
104 using char_type = typename string_type_t::value_type; // Character Type
105 using traits_type = typename string_type_t::traits_type; // Traits Type
106 using size_type = typename string_type_t::size_type; // Size Type
107
108 using fig_char_type = std::vector<string_type_t>; // Figlet char
109 using fig_str_type = std::vector<string_type_t>; // Figlet String
110
111 private: // private typedefs
112 using sstream_type = std::basic_stringstream<char_type>; // Sstream Type
113 using ostream_type = std::basic_ostream<char_type>; // Ostream Type
114 using istream_type = std::basic_istream<char_type>; // Istream Type
115 using ifstream_type = std::basic_ifstream<char_type>; // Ifstream Type
116 using ofstream_type = std::basic_ofstream<char_type>; // Ofstream Type
117
118 protected: // protected methods
122 void trim_deep(fig_str_type &fig_str, fig_char_type &fig_ch) const
123 {
124 // left spaces and right spaces
125 std::vector<size_type> elem;
126
127 // count space
128 for (size_type i = 0; i < fig_str.size(); ++i)
129 {
130 int l_count = 0, r_count = 0;
131
132 for (auto itr = fig_str[i].rbegin(); itr != fig_str[i].rend(); ++itr)
133 {
134 if (*itr == ' ')
135 ++l_count;
136 else
137 break;
138 }
139
140 for (auto itr = fig_ch[i].begin(); itr != fig_ch[i].end(); ++itr)
141 {
142 if (*itr == ' ')
143 ++r_count;
144 else
145 break;
146 }
147
148 elem.push_back(l_count + r_count);
149 }
150
151 // minimum
152 const auto min = *std::min_element(elem.begin(), elem.end());
153
154 // for each line
155 for (size_type i = 0; i < fig_str.size(); ++i)
156 {
157 size_type siz = min + 1;
158
159 while (--siz > 0 && fig_str[i].back() == ' ')
160 {
161 fig_str[i].pop_back();
162 }
163
164 fig_ch[i].erase(0, siz);
165 }
166 }
167
168 public: // Public overrides
172 shrink_type get_shrink_level() const override
173 {
174 return shrink_type::KERNING;
175 }

227 using char_type = typename string_type_t::value_type; // Character Type
228 using traits_type = typename string_type_t::traits_type; // Traits Type
229 using size_type = typename string_type_t::size_type; // Size Type
230
231 using fig_char_type = std::vector<string_type_t>; // Figlet char
232 using fig_str_type = std::vector<string_type_t>; // Figlet String
233
234 private: // private typedefs
235 using sstream_type = std::basic_stringstream<char_type>; // Sstream Type
236 using ostream_type = std::basic_ostream<char_type>; // Ostream Type
237 using istream_type = std::basic_istream<char_type>; // Istream Type

241 protected: // protected methods
249 auto smush_rules(char_type lc, char_type rc) const
250 {
251 //()
252 if (lc == ' ')
253 {
254 return rc;
255 }
256
257 if (rc == ' ')
258 {
259 return lc;
260 }
261
262 //(Equal character smush)
263 if (lc == rc)
264 {
265 return rc;
266 }
267
268 //(Underscores smush)
269 if (lc == '_' && this->cvt("|/\\[]{}()<>").find(rc) != string_type::npos)
270 {
271 return rc;
272 }
273
274 if (rc == '_' && this->cvt("|/\\[]{}()<>").find(lc) != string_type::npos)
275 {
276 return lc;
277 }
278
279 //(Hierarchy Smushing)
280 auto find_class = [](char_type ch)
281 {
282 if (ch == '|')
283 {
284 return 1;
285 }
286
287 if (ch == '/' || ch == '\\')
288 {
289 return 3;
290 }
291
292 if (ch == '[' || ch == ']')
293 {
294 return 4;
295 }
296
297 if (ch == '{' || ch == '}')
298 {
299 return 5;
300 }
301
302 if (ch == '(' || ch == ')')
303 {
304 return 6;
305 }
306
307 return 0;
308 };
309
310 size_type c_lc = find_class(lc);
311 size_type c_rc = find_class(rc);
312
313 if (c_lc > c_rc)
314 {
315 return lc;
316 }
317
318 if (c_rc > c_lc)
319 {
320 return rc;
321 }
322
323 //(Opposite smush)
324 if (lc == '[' && rc == ']')
325 {
326 return '|';
327 }
328
329 if (lc == ']' && rc == '[')
330 {
331 return '|';
332 }
333
334 if (lc == '{' && rc == '}')
335 {
336 return '|';
337 }
338
339 if (lc == '}' && rc == '{')
340 {
341 return '|';
342 }
343
344 if (lc == '(' && rc == ')')
345 {
346 return '|';
347 }
348
349 if (lc == ')' && rc == '(')
350 {
351 return '|';
352 }
353
354 //(Big X smush)
355 if (lc == '/' && rc == '\\')
356 {
357 return '|';
358 }
359
360 if (lc == '\\' && rc == '/')
361 {
362 return 'Y';
363 }
364
365 if (lc == '>' && rc == '<')
366 {
367 return 'X';
368 }
369
370 //(universal smush)
371 return lc;
372 }

380 void do_smush(fig_str_type &fig_str, fig_char_type fig_ch) const
381 {
382 // trim the ends of the fig str and fig char
383 this->trim_deep(fig_str, fig_ch);
384
385 // is smushable
386 bool smushable = true;
387
388 // determine if smushable
389 for (size_type i = 0; i < this->height; ++i)
390 {
391 if ((fig_str[i].back() == this->hard_blank) && !(fig_ch[i].front() == this->hard_blank))
392 {
393 smushable = false; break;
394 }
395 else if (fig_str[i].size() == 0 || fig_str[i].size() == 0)
396 {
397 smushable = false; break;
398 }
399 }
400
401 // do the work
402 if (smushable)
403 {
404 for (size_type i = 0; i < fig_str.size(); ++i)
405 {
406 fig_str[i].back() = this->smush_rules(fig_str[i].back(), fig_ch[i].front());
407 fig_ch[i].erase(fig_ch[i].begin());
408 fig_str[i] += fig_ch[i];
409 }
410 }
411 else
412 {
413 for (size_type i = 0; i < fig_str.size(); ++i)
414 {
415 fig_str[i] += fig_ch[i];
416 }
417 }
418 }
