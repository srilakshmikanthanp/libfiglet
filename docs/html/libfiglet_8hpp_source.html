268 void start(std::basic_istream<char_type, traits_type> &stream)
269 {
270 string_type str;
271 size_type dummy;
272 size_type cmnt;
273 int old_tag;
274
275 stream >> str;
276
277 if (str.substr(0, 5) != cvt("flf2a"))
278 {
279 throw std::runtime_error("flf2a not found !");
280 }
281
282 str.erase(0, 5);
283
284 this->HardBlank = str.at(0); // HardBlank
285
286 stream >> this->Height; // Height
287 stream >> dummy;
288 stream >> dummy;
289 stream >> old_tag;
290 stream >> cmnt; // comment line
291
292 std::getline(stream, str);
293
294 for (size_type i = 0; i < cmnt; i++)
295 {
296 std::getline(stream, str);
297 }
298
299 if (old_tag < 0)
300 {
301 this->Shrink = 0;
302 }
303 else if (old_tag == 0)
304 {
305 this->Shrink = 1;
306 }
307 else
308 {
309 this->Shrink = 2;
310 }
311 }

319 Figc_type next(std::basic_istream<char_type, traits_type> &stream)
320 {
321 std::vector<string_type> ret;
322
323 for (size_type i = 1; i <= this->Height; ++i)
324 {
325 string_type str;
326 std::getline(stream, str);
327
328 if (str.back() == '\n' || str.back() == '\r')
329 {
330 str.pop_back();
331 }
332
333 str.pop_back();
334 ret.push_back(str);
335
336 if (i == this->Height)
337 {
338 ret.back().pop_back();
339 }
340 }
341
342 return ret;
343 }

576 class BasicKerning : public BasicFullWidth<StringType>
577 {
578 public:
579 using string_type = StringType;
580 using char_type = typename string_type::value_type;
581 using traits_type = typename string_type::traits_type;
582 using size_type = typename string_type::size_type;

584 using Figs_type = std::vector<string_type>;
585
586 private:
587 size_type gap;
588
589 protected:
596 void trim_deep(Figs_type &figs, Figc_type &figc) const
597 {
598 std::vector<size_type> elem;
599
600 for (size_type i = 0; i < figs.size(); ++i)
601 {
602 int lcount = 0, rcount = 0;
603
604 for (auto itr = figs[i].rbegin(); itr != figs[i].rend(); ++itr)
605 {
606 if (*itr == ' ')
607 ++lcount;
608 else
609 break;
610 }
611
612 for (auto itr = figc[i].begin(); itr != figc[i].end(); ++itr)
613 {
614 if (*itr == ' ')
615 ++rcount;
616 else
617 break;
618 }
619
620 elem.push_back(lcount + rcount);
621 }
622
623 size_type space = *std::min_element(elem.begin(), elem.end());
624
625 for (size_type i = 0; i < figs.size(); ++i)
626 {
627 size_type siz = space;
628
629 while (siz > 0 && figs[i].back() == ' ')
630 {
631 figs[i].pop_back();
632 --siz;
633 }
634
635 figc[i].erase(0, siz);
636 }
637 }

710 template <class StringType>
711 class BasicSmushed : public BasicKerning<StringType>
712 {
713 public:
714 using string_type = StringType;
715 using char_type = typename string_type::value_type;
716 using traits_type = typename string_type::traits_type;
717 using size_type = typename string_type::size_type;
718 using Figc_type = std::vector<string_type>;
719 using Figs_type = std::vector<string_type>;
720
721 private:
728 char_type smush_rules(char_type lc, char_type rc) const
729 {
730 //()
731 if (lc == ' ')
732 {
733 return rc;
734 }
735
736 if (rc == ' ')
737 {
738 return lc;
739 }
740
741 //(Equal character smush)
742 if (lc == rc)
743 {
744 return rc;
745 }
746
747 //(Underscores smush)
748 if (lc == '_' && this->cvt("|/\\[]{}()<>").find(rc) != string_type::npos)
749 {
750 return rc;
751 }
752
753 if (rc == '_' && this->cvt("|/\\[]{}()<>").find(lc) != string_type::npos)
754 {
755 return lc;
756 }
757
758 //(Hierarchy Smushing)
759 auto find_class = [](char_type ch) -> size_type
760 {
761 if (ch == '|')
762 {
763 return 1;
764 }
765
766 if (ch == '/' || ch == '\\')
767 {
768 return 3;
769 }
770
771 if (ch == '[' || ch == ']')
772 {
773 return 4;
774 }
775
776 if (ch == '{' || ch == '}')
777 {
778 return 5;
779 }
780
781 if (ch == '(' || ch == ')')
782 {
783 return 6;
784 }
785
786 return 0;
787 };
788
789 size_type c_lc = find_class(lc);
790 size_type c_rc = find_class(rc);
791
792 if (c_lc > c_rc)
793 {
794 return lc;
795 }
796
797 if (c_rc > c_lc)
798 {
799 return rc;
800 }
801
802 //(Opposite smush)
803 if (lc == '[' && rc == ']')
804 {
805 return '|';
806 }
807
808 if (lc == ']' && rc == '[')
809 {
810 return '|';
811 }
812
813 if (lc == '{' && rc == '}')
814 {
815 return '|';
816 }
817
818 if (lc == '}' && rc == '{')
819 {
820 return '|';
821 }
822
823 if (lc == '(' && rc == ')')
824 {
825 return '|';
826 }
827
828 if (lc == ')' && rc == '(')
829 {
830 return '|';
831 }
832
833 //(Big X smush)
834 if (lc == '/' && rc == '\\')
835 {
836 return '|';
837 }
838
839 if (lc == '\\' && rc == '/')
840 {
841 return 'Y';
842 }
843
844 if (lc == '>' && rc == '<')
845 {
846 return 'X';
847 }
848
849 //(universel smush)
850 return lc;
851 }
852
859 void smush(Figs_type &figs, Figc_type figc, char_type hb) const
860 {
861 bool smushble = true;
862
863 for (size_type i = 0; i < figs.size(); ++i)
864 {
865 if (figs[i].size() == 0 || figc[i].size() == 0)
866 {
867 smushble = false;
868 }
869 else if ((figs[i].back() == hb) && !(figc[i].front() == hb))
870 {
871 smushble = false;
872 }
873 }
874
875 if (smushble)
876 {
877 for (size_type i = 0; i < figs.size(); ++i)
878 {
879 char_type val = smush_rules(figs[i].back(), figc[i].front());
880 figs[i].pop_back();
881 figc[i].erase(0, 1);
882 figs[i] += string_type(1, val) + figc[i];
883 }
884 }
885 else
886 {
887 for (size_type i = 0; i < figs.size(); ++i)
888 {
889 figs[i] += figc[i];
890 }
891 }
892 }

897 BasicSmushed(BasicSmushed &&) = default;

908 {

924 }
