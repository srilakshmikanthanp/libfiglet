286 void read_config_and_remove_comments(typename fig_types::istream_type &is)
287 {
288 // flf header line container
289 typename fig_types::string_type flf_config_line;
290
291 // get the first line
292 std::getline(is, flf_config_line);
293
294 // string stream for the first line
295 typename fig_types::sstream_type ss(flf_config_line);
296
297 // token
298 typename fig_types::string_type token;
299
300 // Read header
301 ss >> std::setw(5) >> token;
302
303 // check
304 if (token != "flf2a")
305 {
306 throw std::runtime_error("Invalid flf header");
307 }
308
309 // Read hard blank
310 ss >> this->hard_blank;
311
312 // check stream
313 if (ss.fail())
314 {
315 throw std::runtime_error("Invalid flf header");
316 }
317
318 // Read height
319 ss >> this->height;
320
321 // check stream
322 if (ss.fail())
323 {
324 throw std::runtime_error("Invalid flf header");
325 }
326
327 // Read baseline
328 ss >> token;
329
330 // check stream
331 if (ss.fail())
332 {
333 throw std::runtime_error("Invalid flf header");
334 }
335
336 // Read max length
337 ss >> token;
338
339 // check stream
340 if (ss.fail())
341 {
342 throw std::runtime_error("Invalid flf header");
343 }
344
345 // Read old layout
346 ss >> token;
347
348 // check stream
349 if (ss.fail())
350 {
351 throw std::runtime_error("Invalid flf header");
352 }
353
354 // set shrink level
355 const auto old_layout = std::stoi(token);
356
357 if (old_layout < 0) // less than 0 then FULL_WIDTH
358 {
359 this->shrink = fig_types::shrink_type::FULL_WIDTH;
360 }
361
362 if (old_layout == 0) // equal to 0 then KERNING
363 {
364 this->shrink = fig_types::shrink_type::KERNING;
365 }
366
367 if (old_layout > 0) // greater than 0 then SMUSHED
368 {
369 this->shrink = fig_types::shrink_type::SMUSHED;
370 }
371
372 // Read comment lines
373 ss >> token;
374
375 // check and set
376 const auto comment_lines = std::stoi(token);
377
378 // ignore comment lines
379 for (auto i = 0; i < comment_lines; ++i)
380 {
381 std::getline(is, token);
382 }
383
384 // Read print direction
385 ss >> token;
386
387 // check stream
388 if (ss.fail())
389 {
390 throw std::runtime_error("Invalid flf header");
391 }
392
393 // Read full layout
394 ss >> token;
395
396 // check stream
397 if (ss.fail())
398 {
399 throw std::runtime_error("Invalid flf header");
400 }
401
402 // Read codetag count
403 ss >> token;
404
405 // check stream
406 if (ss.fail())
407 {
408 throw std::runtime_error("Invalid flf header");
409 }

415 typename fig_types::fig_char_type str_to_fig_char(const typename fig_types::string_type &str) const
416 {
417 // fig char container
418 typename fig_types::fig_char_type fig_char;
419
420 // string stream
421 typename fig_types::sstream_type ss(str);
422
423 // line
424 typename fig_types::string_type line;
425
426 // read lines
427 while (std::getline(ss, line))
428 {
429 // pop the last char
430 line.pop_back();
431
432 // push the line
433 fig_char.push_back(line);
434 }
435
436 // check height
437 if (fig_char.size() != this->height)
438 {
439 throw std::runtime_error("Invalid flf font");
440 }
441
442 // return
443 return fig_char;

449 void read_chars(typename fig_types::istream_type &is)
450 {
451 // Regular expression for flf char
452 std::regex pattern("(.*(?=[@#])[\\s\\S]*?[@#](?=[@#]))", std::regex_constants::ECMAScript);
453
454 // Read the whole file
455 const auto file_str = typename fig_types::string_type(ibuff_it(is), ibuff_it());
456
457 // get all the matches
458 std::match_results<typename fig_types::string_type::const_iterator> res;
459
460 // start char
461 typename fig_types::char_type ch = ' ';
462
463 // start
464 auto start = file_str.cbegin();
465
466 // end
467 auto end = file_str.cend();
468
469 // read all the characters
470 for (ch = ' '; std::regex_search(start, end, res, pattern) && ch <= '~'; ++ch)
471 {
472 // get the fig char
473 const auto fig_char = this->str_to_fig_char(res[0].str());
474
475 // insert the fig char
476 this->fig_chars.insert({ch, fig_char});
477
478 // update start
479 start = res.suffix().first + 1;
480 }
481
482 // check
483 if (--ch != '~')
484 {
485 throw std::runtime_error("Invalid flf characters");
486 }
487 }

646 template <typename fig_types>
647 struct basic_kerning_style : public basic_full_width_style<fig_types>
648 {
649 protected: // protected methods
653 void trim_deep(typename fig_types::fig_str_type &fig_str, typename fig_types::fig_char_type &fig_ch) const
654 {
655 // left spaces and right spaces
656 std::vector<typename fig_types::size_type> elem;
657
658 // count space
659 for (typename fig_types::size_type i = 0; i < fig_str.size(); ++i)
660 {
661 int l_count = 0, r_count = 0;
662
663 for (auto itr = fig_str[i].rbegin(); itr != fig_str[i].rend(); ++itr)
664 {
665 if (*itr == ' ')
666 ++l_count;
667 else
668 break;
669 }
670
671 for (auto itr = fig_ch[i].begin(); itr != fig_ch[i].end(); ++itr)
672 {
673 if (*itr == ' ')
674 ++r_count;
675 else
676 break;
677 }
678
679 elem.push_back(l_count + r_count);
680 }
681
682 // minimum
683 const auto min = *std::min_element(elem.begin(), elem.end());
684
685 // for each line
686 for (typename fig_types::size_type i = 0; i < fig_str.size(); ++i)
687 {
688 typename fig_types::size_type siz = min + 1;
689
690 while (--siz > 0 && fig_str[i].back() == ' ')
691 {
692 fig_str[i].pop_back();
693 }
694
695 fig_ch[i].erase(0, siz);
696 }
697 }

699 public: // Public overrides
703 typename fig_types::shrink_type get_shrink_level() const override
704 {
705 return fig_types::shrink_type::KERNING;
706 }

756 template <typename fig_types>
757 struct basic_smushed_style : public basic_kerning_style<fig_types>
758 {
759 protected: // private methods
767 auto smush_rules(typename fig_types::char_type lc, typename fig_types::char_type rc) const
768 {
769 //()
770 if (lc == ' ')
771 {
772 return rc;
773 }
774
775 if (rc == ' ')
776 {
777 return lc;
778 }
779
780 //(Equal character smush)
781 if (lc == rc)
782 {
783 return rc;
784 }
785
786 //(Underscores smush)
787 if (lc == '_' && this->cvt("|/\\[]{}()<>").find(rc) != fig_types::string_type::npos)
788 {
789 return rc;
790 }
791
792 if (rc == '_' && this->cvt("|/\\[]{}()<>").find(lc) != fig_types::string_type::npos)
793 {
794 return lc;
795 }
796
797 //(Hierarchy Smushing)
798 auto find_class = [](typename fig_types::char_type ch)
799 {
800 if (ch == '|')
801 {
802 return 1;
803 }
804
805 if (ch == '/' || ch == '\\')
806 {
807 return 3;
808 }
809
810 if (ch == '[' || ch == ']')
811 {
812 return 4;
813 }
814
815 if (ch == '{' || ch == '}')
816 {
817 return 5;
818 }
819
820 if (ch == '(' || ch == ')')
821 {
822 return 6;
823 }
824
825 return 0;
826 };
827
828 typename fig_types::size_type c_lc = find_class(lc);
829 typename fig_types::size_type c_rc = find_class(rc);
830
831 if (c_lc > c_rc)
832 {
833 return lc;
834 }
835
836 if (c_rc > c_lc)
837 {
838 return rc;
839 }
840
841 //(Opposite smush)
842 if (lc == '[' && rc == ']')
843 {
844 return '|';
845 }
846
847 if (lc == ']' && rc == '[')
848 {
849 return '|';
850 }
851
852 if (lc == '{' && rc == '}')
853 {
854 return '|';
855 }
856
857 if (lc == '}' && rc == '{')
858 {
859 return '|';
860 }
861
862 if (lc == '(' && rc == ')')
863 {
864 return '|';
865 }
866
867 if (lc == ')' && rc == '(')
868 {
869 return '|';
870 }
871
872 //(Big X smush)
873 if (lc == '/' && rc == '\\')
874 {
875 return '|';
876 }
877
878 if (lc == '\\' && rc == '/')
879 {
880 return 'Y';
881 }
882
883 if (lc == '>' && rc == '<')
884 {
885 return 'X';
886 }
887
888 //(universal smush)
889 return lc;
890 }

898 void do_smush(typename fig_types::fig_str_type &fig_str,typename fig_types::fig_char_type fig_ch) const
899 {
900 // trim the ends of the fig str and fig char
901 this->trim_deep(fig_str, fig_ch);
902
903 // is smushable
904 bool smushable = true;
905
906 // determine if smushable
907 for (auto &line : fig_str)
908 {
909 if ((line.back() == this->hard_blank) && !(line.front() == this->hard_blank))
910 {
911 smushable = false;
912 }
913 else if (line.size() == 0 || line.size() == 0)
914 {
915 smushable = false;
916 }
917 }
918
919 if (smushable)
920 {
921 for (typename fig_types::size_type i = 0; i < fig_str.size(); ++i)
922 {
923 fig_str[i].back() = this->smush_rules(fig_str[i].back(), fig_ch[i].front());
924 fig_ch[i].erase(fig_ch[i].begin());
925 fig_str[i] += fig_ch[i];
926 }
927 }
928 else
929 {
930 for (typename fig_types::size_type i = 0; i < fig_str.size(); ++i)
931 {
932 fig_str[i] += fig_ch[i];
933 }
934 }
935 }
