66 fig_char_type to_fig_char(const std::vector<string_type> &lines) const
67 {
68 // regex pattern to match
69 std::regex pattern("(.)\\1?\n", std::regex::ECMAScript);
70
71 // fig char container
72 fig_char_type fig_char;
73
74 // line
75 string_type line;
76
77 // read lines
78 for (const auto &line : lines)
79 {
80 fig_char.push_back(std::regex_replace(line + "\n", pattern, ""));
81 }
82
83 // check height
84 if (fig_char.size() != this->height)
85 {
86 throw std::runtime_error("Height not match");
87 }
88
89 // return
90 return fig_char;
91 }

96 void read_config_and_remove_comments(istream_type &is)
97 {
98 // flf header line container
99 string_type flf_config_line;
100
101 // get the first line
102 std::getline(is, flf_config_line);
103
104 // string stream for the first line
105 sstream_type ss(flf_config_line);
106
107 // token
108 string_type token;
109
110 // Read header
111 ss >> std::setw(5) >> token;
112
113 // check
114 if (token != "flf2a")
115 {
116 throw std::runtime_error("Invalid flf2a header");
117 }
118
119 // Read hard blank
120 ss >> this->hard_blank;
121
122 // check stream
123 if (ss.fail())
124 {
125 throw std::runtime_error("Invalid hard blank");
126 }
127
128 // Read height
129 ss >> this->height;
130
131 // check stream
132 if (ss.fail())
133 {
134 throw std::runtime_error("Invalid height");
135 }
136
137 // Read baseline
138 ss >> token;
139
140 // check stream
141 if (ss.fail())
142 {
143 throw std::runtime_error("Invalid baseline");
144 }
145
146 // Read max length
147 ss >> token;
148
149 // check stream
150 if (ss.fail())
151 {
152 throw std::runtime_error("Invalid max length");
153 }
154
155 // Read old layout
156 ss >> token;
157
158 // check stream
159 if (ss.fail())
160 {
161 throw std::runtime_error("Invalid old layout");
162 }
163
164 // set shrink level
165 const auto old_layout = std::stoi(token);
166
167 if (old_layout < 0) // less than 0 then FULL_WIDTH
168 {
169 this->shrink = shrink_type::FULL_WIDTH;
170 }
171
172 if (old_layout == 0) // equal to 0 then KERNING
173 {
174 this->shrink = shrink_type::KERNING;
175 }
176
177 if (old_layout > 0) // greater than 0 then SMUSHED
178 {
179 this->shrink = shrink_type::SMUSHED;
180 }
181
182 // Read comment lines
183 ss >> token;
184
185 // check stream
186 if (ss.fail())
187 {
188 throw std::runtime_error("Invalid comment lines");
189 }
190
191 // check and set
192 const auto comment_lines = std::stoi(token);
193
194 // ignore comment lines
195 for (auto i = 0; i < comment_lines; ++i)
196 {
197 std::getline(is, token);
198 }
199 }

205 auto read_lines(istream_type &is, size_type n) const
206 {
207 // lines container
208 std::vector<string_type> lines;
209
210 // line
211 string_type line;
212
213 // read lines
214 for (auto i = 0; i < n && std::getline(is, line); ++i)
215 {
216 lines.push_back(line);
217 }
218
219 // return
220 return lines;
221 }

226 void read_chars(istream_type &is)
227 {
228 // read all the characters (ch <= '~' must be first)
229 for (char_type ch = ' '; ch <= '~'; ++ch)
230 {
231 // get the fig char
232 const auto fig_char = this->to_fig_char(this->read_lines(is, this->height));
233
234 // if length less
235 if (fig_char.size() < this->height)
236 {
237 throw std::runtime_error("Invalid fig char");
238 }
239
240 // insert the fig char
241 this->fig_chars.insert({ch, fig_char});
242 }
