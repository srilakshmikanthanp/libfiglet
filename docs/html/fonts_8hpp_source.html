67 void read_config_and_remove_comments(istream_type &is)
68 {
69 // flf header line container
70 string_type flf_config_line;
71
72 // get the first line
73 std::getline(is, flf_config_line);
74
75 // string stream for the first line
76 sstream_type ss(flf_config_line);
77
78 // token
79 string_type token;
80
81 // Read header
82 ss >> std::setw(5) >> token;
83
84 // check
85 if (token != cvt<string_type>("flf2a"))
86 {
87 throw std::runtime_error("Invalid flf2a header");
88 }
89
90 // Read hard blank
91 ss >> this->hard_blank;
92
93 // check stream
94 if (ss.fail())
95 {
96 throw std::runtime_error("Invalid hard blank");
97 }
98
99 // Read height
100 ss >> this->height;
101
102 // check stream
103 if (ss.fail())
104 {
105 throw std::runtime_error("Invalid height");
106 }
107
108 // Read baseline
109 ss >> token;
110
111 // check stream
112 if (ss.fail())
113 {
114 throw std::runtime_error("Invalid baseline");
115 }
116
117 // Read max length
118 ss >> token;
119
120 // check stream
121 if (ss.fail())
122 {
123 throw std::runtime_error("Invalid max length");
124 }
125
126 // Read old layout
127 ss >> token;
128
129 // check stream
130 if (ss.fail())
131 {
132 throw std::runtime_error("Invalid old layout");
133 }
134
135 // set shrink level
136 const auto old_layout = std::stoi(token);
137
138 if (old_layout < 0) // less than 0 then FULL_WIDTH
139 {
140 this->shrink = shrink_type::FULL_WIDTH;
141 }
142
143 if (old_layout == 0) // equal to 0 then KERNING
144 {
145 this->shrink = shrink_type::KERNING;
146 }
147
148 if (old_layout > 0) // greater than 0 then SMUSHED
149 {
150 this->shrink = shrink_type::SMUSHED;
151 }
152
153 // Read comment lines
154 ss >> token;
155
156 // check stream
157 if (ss.fail())
158 {
159 throw std::runtime_error("Invalid comment lines");
160 }
161
162 // check and set
163 const auto comment_lines = std::stoi(token);
164
165 // ignore comment lines
166 for (auto i = 0; i < comment_lines; ++i)
167 {
168 std::getline(is, token);
169 }
170 }

175 void read_chars(istream_type &is)
176 {
177 // lambda function to convert the lines to fig char
178 auto to_fig_char = [this](const std::vector<string_type> &lines) {
179 // regex pattern to match
180 std::basic_regex<char_type> pattern(cvt<string_type>("(.)\\1?\n"), std::regex::ECMAScript);
181
182 // fig char container
183 fig_char_type fig_char;
184
185 // line
186 string_type line;
187
188 // read lines
189 for (const auto &line : lines)
190 {
191 fig_char.push_back(std::regex_replace(line + cvt<string_type>("\n"), pattern, cvt<string_type>("")));
192 }
193
194 // check height
195 if (fig_char.size() != this->height)
196 {
197 throw std::runtime_error("Height not match");
198 }
199
200 // return
201 return fig_char;
202 };
203
204 // lambda function to read lines from the stream
205 auto read_lines = [this](istream_type &is, size_type n) {
206 // lines container
207 std::vector<string_type> lines;
208
209 // line
210 string_type line;
211
212 // read lines
213 for (auto i = 0; i < n && std::getline(is, line); ++i)
214 {
215 lines.push_back(line);
216 }
217
218 // return
219 return lines;
220 };
221
222 // read all the characters (ch <= '~' must be first)
223 for (char_type ch = ' '; ch <= '~'; ++ch)
224 {
225 // get the fig char
226 const auto fig_char = to_fig_char(read_lines(is, this->height));
227
228 // if length less
229 if (fig_char.size() < this->height)
230 {
231 throw std::runtime_error("Invalid fig char");
232 }
233
234 // insert the fig char
235 this->fig_chars.insert({ch, fig_char});
236 }
237 }
